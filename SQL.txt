1. SELECT DEPARTMENT.NAME, EMPLOYEE.NAME
FROM
    DEPARTMENT INNER JOIN EMPLOYEE
    ON DEPARTMENT.ID = EMPLOYEE.DEP_ID
WHERE EMPLOYEE.CHIEF_ID is null

2. select NAME, DEP_ID from EMPLOYEE
where SUBSTRING(EMPLOYEE.NAME, 1, CHARINDEX(' ', EMPLOYEE.NAME) - 1) in
      (select SUBSTRING(EMPLOYEE.NAME, 1, CHARINDEX(' ', EMPLOYEE.NAME) - 1)
      from EMPLOYEE
      group by SUBSTRING(EMPLOYEE.NAME, 1, CHARINDEX(' ', EMPLOYEE.NAME) - 1)
      having count(*) > 1)
GROUP BY NAME, DEP_ID
